-- 問7
SELECT
	pr.name AS prefecture,
	p.gender_code AS gender,
	CASE p.gender_code WHEN 'm' THEN '男'
					 WHEN 'f' THEN '女'
	END,
	sum(p.population) AS population
FROM populations AS p INNER JOIN prefectures AS pr ON p.prefecture_code = pr.code GROUP BY pr.name, p.gender_code, p.prefecture_code;
